# word-kit usage examples

Concrete patterns for common workflows. Every snippet imports from
`@word-kit/core`. They are runnable as-is in Node ≥ 20 once the package
is installed.

---

## 1. Generate a fresh `.docx` from scratch

```ts
import { Docx, PAGE_SIZE_A4 } from "@word-kit/core";
import { writeFileSync } from "node:fs";

const doc = Docx.create({ paragraphs: [] });

doc.setPageSize(PAGE_SIZE_A4);
doc.setCoreProperties({
  title: "Quarterly Report",
  creator: "Yamada",
  created: new Date().toISOString(),
});

doc.ensureHeadingStyles(2);
doc.appendHeading("Quarterly Report", 1);
doc.appendParagraph("This document was generated by word-kit.");

doc.addBulletList(["Revenue ↑ 12%", "Costs ↓ 3%"]);

doc.addTable([
  ["Metric", "Q3"],
  ["Users", "2,200"],
  ["MRR", "$24k"],
]);

doc.addPageNumberFooter("Page ", "");

writeFileSync("out.docx", doc.toUint8Array());
```

## 2. Fill a Word template (run-spanning safe)

```ts
import { Docx } from "@word-kit/core";
import { readFileSync, writeFileSync } from "node:fs";

const tpl = Docx.open(readFileSync("template.docx"));

// Replace across body + headers + footers + comments + footnotes.
const values: Record<string, string> = {
  name: "山田太郎",
  date: new Date().toLocaleDateString("ja-JP"),
  amount: "¥1,200,000",
};
tpl.replaceTextEverywhere(/\{\{\s*(\w+)\s*\}\}/g, (m) => values[m.captures[0]!] ?? "");

writeFileSync("filled.docx", tpl.toUint8Array());
```

## 3. Mail-merge / fan-out one template into N documents

```ts
import { Docx } from "@word-kit/core";
import { readFileSync, writeFileSync } from "node:fs";

const template = Docx.open(readFileSync("template.docx"));

for (const customer of customers) {
  const doc = template.clone();
  doc.replaceTextEverywhere(/\{\{(\w+)\}\}/g, (m) => customer[m.captures[0]!] ?? "");
  doc.setCoreProperties({
    title: `Invoice for ${customer.name}`,
    creator: "Acme Billing",
    modified: new Date().toISOString(),
  });
  writeFileSync(`out/${customer.id}.docx`, doc.toUint8Array());
}
```

`docx.clone()` is a deep copy. Edits do not leak across siblings.

## 4. Inspect an existing `.docx`

```ts
import { Docx } from "@word-kit/core";

const doc = Docx.open(bytes);

doc.statistics;
// { paragraphs: 42, tables: 3, images: 5, comments: 4, footnotes: 2,
//   endnotes: 0, headings: 8, words: 1240, characters: 7820,
//   charactersNoSpaces: 6480 }

doc.outline();
// [ { paragraph, level: 1, text: "Chapter 1" }, { level: 2, ... } ]

doc.bookmarks; // [{ name, id, paragraph }]
doc.headers; // [{ partName, relId, text }]
doc.footers; // [{ partName, relId, text }]
doc.images; // [{ partName, contentType, data }]
doc.coreProperties; // { title?, creator?, subject?, ... }

// Spot template placeholders before bulk-replacing:
for (const { partName, matches } of doc.findTextEverywhere(/\{\{\w+\}\}/g)) {
  console.log(
    partName,
    matches.map((m) => m.text),
  );
}
```

## 6. Add an image, replace an image

```ts
import { Docx } from "@word-kit/core";
import { readFileSync } from "node:fs";

const doc = Docx.create({ paragraphs: ["Logo:"] });
const logo = readFileSync("logo.png");
doc.addImage(logo, {
  widthEmu: 914400,
  heightEmu: 914400,
  altText: "Company logo",
});

// Replace an existing image in a template without redoing rels:
const tpl = Docx.open(readFileSync("template.docx"));
const firstImage = tpl.images[0];
if (firstImage) {
  tpl.replaceImage(firstImage.partName, readFileSync("new-logo.png"));
}
```

## 7. Validate before saving

```ts
import { Docx } from "@word-kit/core";

const doc = buildSomeDocument();
const issues = doc.validate();
if (issues.some((i) => i.level === "error")) {
  console.error("Refusing to save — would produce a docx that Word repairs:");
  for (const issue of issues) console.error(` [${issue.level}] ${issue.code} ${issue.message}`);
  process.exit(1);
}
```

## 8. Browser: load via `<input type=file>` and save via `<a download>`

```ts
import { Docx } from "@word-kit/core";

fileInput.addEventListener("change", async () => {
  const file = fileInput.files?.[0];
  if (!file) return;
  const doc = await Docx.fromBlob(file);
  doc.replaceText("{{name}}", "Alice");

  const blob = doc.toBlob();
  const a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "filled.docx";
  a.click();
  URL.revokeObjectURL(a.href);
});
```
